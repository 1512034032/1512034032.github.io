    局部组件 组件嵌套组件 父组件向子组件传值

    <template id="fu"> // 创建组件内容 根据 id 作为唯一指定
    <div>
        <h2>父组件</h2>
        <p>{{title}}</p>
        <v-zi :name='title'></v-zi><!-- 嵌套子组件 利用自定义标签属性名 name 并给子组件绑定传递的数据 -->
    </div>
</template>
<template id="zi">
    <div>
        <h2>子组件</h2>
        <p>{{name}}</p> <!-- 使用 props 接受到的数据 -->
    </div>
</template>

<script type="text/javascript">
    new Vue({
        el:'#app',
        components:{ // 注册组件
            'v-fu':{ // 组件名称
                template:'#fu', //指定组件的内容区
                data(){ // 当前组件区域内的数据
                    return {title:'Yo~'}
                },
                methods:{
                    // ...
                },
                // 局部创建子组件 必须写在父组件的内部 全局的不用
                components:{ // 子组件注册 -> 嵌套 
                    'v-zi':{
                        template:'#zi',
                        props:['name'], // 获取父组件根据标签属性名传入的数据
                        data(){
                            return {
                                // ...
                            }
                        },
                        methods:{
                            // ....
                        }
                    }
                }
            }
        }
    })
</script>

    组件 子组件向父组件传值

    <div id="app">
    <v-parent></v-parent>
</div>
<template id="parent">
    <div>
        <h2>父组件</h2>
        <p>{{str}}</p>
        <v-child @toparent='gotoStr'></v-child> <!-- 接收子组件传值定义的事件名 toparent 用来绑定函数 -->
    </div>
</template>
<template id="child">
    <div>
        <h2>子组件</h2>
        <p>{{title}}</p>
        <button type="button" @click="tit">btn</button> <!-- 子组件中定义一个方法 -->
    </div>
</template>
<script type="text/javascript">
    new Vue({
        el: '#app',
        components: {
            'v-parent': {
                template: '#parent',
                data() {
                    return {
                        str: ''
                    }
                },
                methods:{
                    gotoStr(msg){
                        this.str = msg // 参数就是接收到值
                    }
                },
                components: {
                    'v-child': {
                        template: '#child',
                        data() {
                            return {
                                title: 'Yo.~' // 子组件的数据
                            }
                        },
                        methods: {
                            tit() {
                                this.$emit('toparent', this.title) // 利用 $emit 创建事件 toparent 和要传递的数据
                            }
                        }
                    }
                }
            }
        }
    })
</script>

    同级别组件之间传值

    <div id="app">
    <v-a></v-a>
    <hr>
    <v-b></v-b>
    <hr>
    <v-c></v-c>
</div>

<template id="a">
    <div>
        <h2>A组件</h2>
        <p>{{title}}</p>
        <button type="button" @click="send">发送数据</button> <!-- 绑定发送数据的方法 -->
    </div>
</template>

<template id="b">
    <div>
        <h2>B组件</h2>
        <p>{{title}}</p>
        <button type="button" @click="send">发送数据</button> <!-- 绑定发送数据的方法 -->
    </div>
</template>

<template id="c">
    <div>
        <h2>C组件</h2>
        <p>接收AAA数据：{{stra}}</p> <!-- 使用接收到的数据 -->
        <p>接收BBB数据：{{strb}}</p>
    </div>
</template>

<script type="text/javascript">
    var vm = new Vue() // 在外部新创建一个 Vue 实例 用来接收传递的值
    new Vue({
        el: '#app',
        components: {
            'v-a': {
                template: '#a',
                data() {
                    return {
                        title: 'a数据'
                    }
                },
                methods: {
                    send() {
                        vm.$emit('isa', this.title) //把数据传递给外部的 Vue 实例
                    }
                }
            },
            'v-b': {
                template: '#b',
                data() {
                    return {
                        title: 'b数据'
                    }
                },
                methods: {
                    send() {
                        vm.$emit('isb', this.title) //把数据传递给外部的 Vue 实例
                    }
                }
            },
            'v-c': {
                template: '#c',
                data() {
                    return {
                        stra: '',
                        strb: ''
                    }
                },
                mounted(){
                    vm.$on('isa',(msg)=>{ // 接收 vm 实例获中的数据 参数表示获取到的数据
                        this.stra = msg
                        console.log(msg)
                    })

                    vm.$on('isb',(msg)=>{
                        this.strb = msg
                    })
                }
            }
        }
    })
</script>

    组件之间的互相操作

    父操作子组件

    <div id="app">
    <h2>父操作子组件</h2>
    <v-parent></v-parent>
</div>
<template id="parent">
    <div>
        <h2>父组件</h2>
        <p>{{title}}</p>
        <button type="button" @click='getchild'>获取子组件</button> <!-- 创建方法用于获取子组件 -->
        <hr>
        <v-child ref='childs'></v-child> <!-- 利用 $refs中的 ref 先给子组件绑定获取时用到的名称 -->
    </div>
</template>

<template id="child">
    <div>
        <h2>子组件</h2>
        <p>{{title}}</p>
    </div>
</template>

<script type="text/javascript">
    new Vue({
        el: '#app',
        components: {
            'v-parent': {
                template: '#parent',
                data() {
                    return {
                        title: 'Yo.~'
                    }
                },
                methods:{
                    getchild(){
                        console.log(this.$refs.childs.$el) // this.$refs.名称 为整个子组件    $el 为子组件的 DOM
                        this.$refs.childs.title='Yo.~2' // 直接可操作子组件的任何东西 包括数据 方法 等等...
                    }
                },
                components: {
                    'v-child': {
                        template: '#child',
                        data() {
                            return {
                                title: 'Yo.~'
                            }
                        }
                    }
                }
            }
        }
    })
</script>

    $nextTick 用于获取修改后的最新 DOM 实例模板

    <div id="app">
    <h2>nextTick</h2>
    <p ref='tit'>{{title}}</p>
</div>

<script type="text/javascript">
    new Vue({
        el:'#app',
        data:{
            title:'Yo.~'
        },
        mounted(){
            this.title = 'Yo2.~'
            console.log(this.$refs.tit.innerHTML) 
            //虽然页面渲染的事最新的数据 但是 这个时候 Vue 生成的 DOM 模板还是旧的 输出 Yo.~
            this.$nextTick(()=>{
                console.log(this.$refs.tit.innerHTML)
                //利用 $nextTick 这个时候 Vue 生成的 DOM 模板是最新的 输出 Yo.2~
            })
        }
    })
</script>

    slot占位标签

    <div id="app">
    <v-head>
        <!-- 1.如果组件中没有 slot 占位标签 自定义标签中写入任何东西都将不生效
             2.在自定义标签中写入标签 可使用 slot 属性来区分要是用的占位标签 -->
        <ul slot='ul'>
            <li>1</li>
        </ul>
    </v-head>
</div>

<template id="head">
    <div>
        <h2>组件</h2>
        <slot name="ul"></slot> 
        <!-- 在组件中生成一个 slot 占位标签 放置多个 可食用 name 属性来区分 -->
    </div>
</template>

<script type="text/javascript">
    new Vue({
        el:'#app',
        components:{
            'v-head':{
                template:'#head'
            }
        }
    })
</script>

    Vue 自定义指令

    全局

    Vue.directive('color',{ //全局生成自定义指令
    // 1.第一个参数表示指令名称 color，生成指定的时候不需要加上 v-，使用方法<p v-color></p>
    // 2.第二个参数 可设置指令在哪个阶段执行
    // 3.1.方法中的第一个参数表示 绑定指定的 DOM 元素
    // 3.2.方法中的第二个参数表示指令上自带的一些数据值 如 v-colo.abc、v-colo='red' 上的 abc 、red等
    inserted:function(el,a){
        el.style.color = 'red'
        console.log(a)
    }
    bind:function(el){
        // 每当指令绑定到元素上的时候,会执行这个 bind 函数,只执行一次
        // 注意:在每个 函数中,第一个参数,永远是 el ,表示被绑定了指令的那个元素,这个 el 参数,是一个原生的Js对象,
        // 在元素 刚绑定指令的时候,喊没有插入带 DOM 中去,这时候 调用 focus 方法没有作用
        // 因为,一个元素,只有插入 DOM 之后,才能获取焦点
    },
    inserted:function(el){ // inserted 表示元素 插入到 DOM 中的时候,会执行 inserted 函数 [指触发一次]
        el.focus()
    },
    update:function(el){ // 当 VNode 更新的时候,会执行 upddate ,可能会触发多次
    }
})
